--Create table

CREATE TABLE CustomerTable 
(   sdCustomer_id Number(3,0) NOT NULL
    , customerEmailAddress varchar2(120) NOT NULL 
    , customerFirstName varchar2(25) NOT NULL
    , customerLastName varChar2(50) NOT NULL    
    , customerStreetAddress varChar2(100) NOT NULL
    , customerCity varChar2(25) NOT NULL
    , customerState varChar2(2) NOT NULL
    , customerZip varChar2(5) NOT NULL );  
CREATE TABLE Supplier
(   sdSupplier_id Number(3,0) NOT NULL
    , supplierName varchar2(100) NOT NULL    
    , supplierStreetAddress varChar2(100) NOT NULL
    , supplierCity varChar2(25) NOT NULL
    , supplierState varChar2(2) NOT NULL
    , supplierZip varChar2(5) NOT NULL );
CREATE TABLE Vendor
(   sdVendor_id Number(3,0) NOT NULL
    , vendorEmailAddress varchar2(120) NOT NULL
    , vendorName varchar2(100) NOT NULL
    , vendorStreetAddress varChar2(100) NOT NULL
    , vendorCity varChar2(25) NOT NULL
    , vendorState varChar2(2) NOT NULL
    , vendorZip varChar2(5) NOT NULL
    , vendorPhone varchar2(10) NOT NULL );
CREATE TABLE ProductTable
(   sdProduct_id Number(3,0) NOT NUll
    , productName varchar2(200) NOT NULL
    , sdSupplier_id Number(3,0) NOT NULL );
CREATE TABLE VendorProduct
(   sdVendorProduct_id Number(3,0) NOT NULL
    , sdVendor_id Number(3,0) 
    , sdProduct_id Number(3,0) 
    , quantityOnHand Number(3,0) 
    , vendorProductPrice Number(6,2) NOT NULL );
CREATE TABLE OrderTable
(   sdOrder_id Number(3,0) NOT NULL
    , sdCustomer_id Number(3,0) NOT NULL
    , orderDateTime Date NOT NULL
    , subTotal Number(6,2) NOT NULL
    , taxAmount Number(6,2) NOT NULL
    , shippingCost Number(6,2) NOT NULL
    , ordertotal Number(6,2) NOT NULL );

CREATE TABLE OrderItem
(   sdOrder_id Number(3,0) NOT NULL
    , sdVendorProduct_id Number(3,0) NOT NULL
    , quantity Number(3,0) NOT NULL );
